CREATE DATABASE bancojava;

USE bancojava;

CREATE TABLE IF NOT EXISTS Usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    cpf VARCHAR(15) NOT NULL UNIQUE,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    saldo double NOT NULL,
    senha VARCHAR(15) NOT NULL,
    senha4digitos INT NOT NULL,
    numeroConta INT NOT NULL UNIQUE
);

CREATE TABLE IF NOT EXISTS Extrato (
    id INT AUTO_INCREMENT PRIMARY KEY,
    cpf VARCHAR(15) NOT NULL,
    data_horario VARCHAR(100) NOT NULL,
    descricao VARCHAR(100) NOT NULL,
    valor double NOT NULL,
    tipo VARCHAR(15) NOT NULL
);

CREATE TABLE IF NOT EXISTS lista_bancos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nomeBanco VARCHAR(200) NOT NULL,
    codigoBanco CHAR(3) NOT NULL
);
